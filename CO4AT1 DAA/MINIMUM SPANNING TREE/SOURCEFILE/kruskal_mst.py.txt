# Kruskal's Algorithm

graph = [
    [0, 2, 3, 0],
    [2, 0, 15, 2],
    [3, 15, 0, 13],
    [0, 2, 13, 0]
]

edges = [
    (2, 0, 1),
    (3, 0, 2),
    (15, 1, 2),
    (2, 1, 3),
    (13, 2, 3)
]

parent = [0, 1, 2, 3]

def find(x):
    while parent[x] != x:
        x = parent[x]
    return x

cost = 0

for w, u, v in sorted(edges):
    if find(u) != find(v):
        parent[find(u)] = find(v)
        cost += w

print("MST Cost =", cost)
